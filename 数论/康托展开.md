# 康托展开

现在有"abcdefghijkl”12个字符，将其所有的排列中按字典序排列，给出任意一种排列，说出这个排列在所有的排列中是第几小的？

样例输入

```
3
abcdefghijkl
hgebkflacdji
gfkedhjblcia
```

样例输出

```
1
302715242
260726926
```

```cpp
char s[15];  
//打表记录阶乘  
const int jc[13]= {1,1,2,6,24,120,720,5040,40320,362880,3628800,39916800,479001600};  
int work()  
{  
    int ans=0;  
    for(int i=0; i<12; i++)  
    {  
        int tmp=0;  
        for(int j=i+1; j<12; j++)  
            if(s[j]<s[i])  
                tmp++;//是第几大的元素  
        ans+=tmp*jc[12-i-1];  
    }  
    return ans;  
}  
int main()  
{  
    int t;  
    scanf("%d",&t);  
    while(t--)  
    {  
        scanf("%s",s);  
        printf("%d\n",work()+1);  
    }  
    return 0;  
}  
```

```cpp
char s[15];  
LL num[20];  
LL jc(LL n)  
{  
    if(n==1)return 1;  
    return n*jc(n-1);  
}  
void init()//打表阶乘  
{  
    for(LL i=1;i<=18;i++)  
        num[i]=jc(i);  
}  
LL work(char s[],LL n)  
{  
    LL ans=0;  
    for(LL i=0; i<n; i++)  
    {  
        LL tmp=0;  
        for(LL j=i+1; j<n; j++)  
            if(s[j]<s[i])  
                tmp++;//是第几大的元素  
        ans+=tmp*num[n-i-1];  
    }  
    return ans;  
}  
int main()  
{  
    init();  
    char s[20]="bckfqlajhemgiodnp";  
    printf("%lld\n",work(s,17));  
    return 0;  
}
```
